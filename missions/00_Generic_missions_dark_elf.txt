	gen_dark_elf_get_slaves = { 
		icon = mission_dark_elf_culture
		required_missions = {  }
		provinces_to_highlight = {
			trade_goods = slaves
		}
		trigger = {
			OR = {
				development_in_provinces = {
					value = 100
					owned_by = ROOT 
					trade_goods = slaves 
				}
				num_of_owned_provinces_with = {
					trade_goods = slaves
					value = 14
				}
			}
		}

		effect = {
			every_owned_province = {
				limit = { trade_goods = slaves }
				add_base_manpower = 1
				add_base_production = 1
			}
			add_years_of_owned_provinces_manpower = {
				years = 4
				trigger = {
					trade_goods = slaves
				}
			}		
		}
	}

	gen_dark_elf_lead_slaves = { 
		icon = mission_dark_elf_culture

			}			
		}
	}

	gen_dark_elf_consolidate_power = {

	slot = 3
	generic = yes
	has_country_shield = no
	ai = yes
	potential = {
		culture_group = druchii_group
	}

	gen_dark_elf_ultimate = { 
		icon = mission_dark_elf_culture
		required_missions = { }
		position = 9
		provinces_to_highlight = {
			OR = {
				continent = north_america
				continent = oceania
			}
			NOT = { owned_by = ROOT }
		}
		trigger = {
			NOT = {
				any_province = {
					OR = {
						continent = north_america
						continent = oceania
					}
					NOT = { owned_by = ROOT }
				}
			}
		}

		effect = {
			custom_tooltip = claim_all_world_tooltip

			add_country_modifier = { 
 				name = dark_elf_dominion
				duration = -1
			}

			hidden_effect = {

				every_province = {
					add_permanent_claim = ROOT
				}
			}
		}
	}
}
