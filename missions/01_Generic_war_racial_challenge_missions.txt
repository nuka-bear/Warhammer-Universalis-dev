race_relations_1_missions = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		always = yes
	}
	### Humans ###
	challenge_humans = {
		icon = mission_cannons_firing
		required_missions = {
		}
		position = 10
		trigger = {
			any_country = {
				has_country_modifier = human_military
				ROOT = {
					has_won_war_against = {
						who = PREV
						max_years_since = 100
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "challenged_humans"
				duration = -1
			}
		}
	}
	trade_humans = {
		icon = mission_war_chest
		required_missions = {
		}
		trigger = {
			any_country = {
				has_country_modifier = human_administration
				trust = {
					who = ROOT
					value = 80
				}
				has_opinion = {
					who = ROOT
					value = 150
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "traded_humans"
				duration = -1
			}
		}
	}
