gen_dwarf_missions = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		OR = {
			culture_group = dwarf_group
			culture_group = chaos_dwarf_group
		}
	}

	gen_dwarf_please_nobility = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			estate_loyalty = {
			    estate = estate_nobles
			    loyalty = 70
			}
		}

		effect = {
			define_general = {
			    shock = 5
			    fire = 6
			    manuever = 2
			    siege = 6
			}
		}
	}

	gen_dwarf_fortify = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_dwarf_please_nobility } 
		trigger = {
			num_of_owned_provinces_with = {
				fort_level = 2
				value = 4
			}
		}

		effect = {
			add_country_modifier = { 
 				name = gen_dwarf_fort_network
				duration = -1
			}
		}
	}

	gen_dwarf_metals = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			copper = 1
			iron = 2
		}

		effect = {
			add_country_modifier = { 
 				name = gen_dwarf_metals_bonus
				duration = -1
			}

			every_province = {
				limit = {
					OR = {
						trade_goods = copper
						trade_goods = iron
					}
				}
				add_permanent_claim = ROOT
			custom_tooltip = claim_all_metals_tooltip
			}	
		}
	}

	gen_dwarf_gold_beer_gems = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			gold = 1			
			beer = 1
			gems = 1
		}

		effect = {
			add_country_modifier = { 
 				name = gen_dwarf_luxury_bonus
				duration = -1
			}

			every_province = {
				limit = {
					OR = {
						trade_goods = gold
						trade_goods = beer
						trade_goods = gems
					}
				}
				add_permanent_claim = ROOT
				custom_tooltip = claim_all_luxury_goods_tooltip
			}
			
		}
	}


	gen_dwarf_make_money_1 = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  }
		position = 6 
		trigger = {
			monthly_income = 25
		}

		effect = {

			random_owned_province = {
				limit = { NOT = { province_has_center_of_trade_of_level = 2 } }

			}

	}

	gen_dwarf_make_money_2 = {

### Thungni is the god of Runic magic and runesmithing.
	gen_ancestor_thungni = {

			}

	}
