gen_asur_missions = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		culture_group = asur_group
	}


#### Send out diplomatic missions

	gen_asur_please_nobility = {
		icon = mission_egyptian_mamluk_soldier
		required_missions = {
		}
		trigger = {
			estate_loyalty = {
				estate = estate_nobles
				loyalty = 65
			}
			num_of_estate_agendas_completed = {
				estate = estate_nobles
				value = 1
			}
		}
		effect = {
			custom_tooltip = asur_diplomatic_missions_tooltip
			
			create_conquistador = { tradition = 50 }

			hidden_effect = {
				every_country = {
					limit = { has_discovered = ROOT }
					add_trust = {
					    who = ROOT
					    value = 5
					}
					add_favors = {
					    who = ROOT
					    value = 5
					}
				}
			}
		}

	}

	gen_asur_get_crownland_up = {
		icon = mission_egyptian_mamluk_soldier
		required_missions = {
			gen_asur_please_nobility
		}
		trigger = {
			crown_land_share = 50
		}
		effect = {
			capital_scope = {
				add_base_tax = 2
				add_base_production = 2
				add_base_manpower = 2
			}
		}
	}

	gen_asur_fight_religious_enemy = {
		icon = mission_egyptian_mamluk_soldier
		required_missions = {
		}
		trigger = {

### Change this to has won war against religious enemy

			at_war_with_religious_enemy = yes
		}
		effect = {
			add_country_modifier = { 
			    name = asur_warscore_against_religious_enemies
			    duration = -1
			}
		}
	}


	gen_asur_enemies_and_alliances = {
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_asur_fight_religious_enemy }
		trigger = {
			any_ally = { NOT = { technology_group = ROOT } }
			any_rival_country = { NOT = { technology_group = ROOT } }
		}
		effect = {
			add_country_modifier = { 
			    name = asur_speedy_diplomats
			    duration = -1
			}
		}
	}



	## Big Fleet

	gen_asur_reclaim_phoenix_crown = {
		icon = mission_egyptian_mamluk_soldier
		required_missions = {
		}
		provinces_to_highlight = {
			has_province_modifier = location_of_the_phoenix_crown
		}
		trigger = {
			any_owned_province = {
				has_province_modifier = location_of_the_phoenix_crown
			}
		}
		effect = {
			add_legitimacy = 50
			add_prestige_or_monarch_power = { amount = 50 }

			custom_tooltip = reclaimed_phoenix_crown_mission_tooltip

			hidden_effect = {
				every_country = {
					limit = {
						OR = {
							culture_group = asur_group
							culture_group = druchii_group
							culture_group = asrai_group							
						}
					}
					add_trust = {
					    who = ROOT
					    value = 20
					}
					add_favors = {
					    who = ROOT
					    value = 20
					}
				}
			}
		}
	}

gen_asur_ultimate_missions = {
	slot = 3
	generic = yes
	ai = yes
	potential = {
		culture_group = asur_group
	}

	gen_asur_ultimate_mission = {
		icon = mission_egyptian_mamluk_soldier
		provinces_to_highlight = {
			owner = {
				NOT = { great_power_rank = 9 }
				NOT = {
					OR = {
						AND = {
							alliance_with = ROOT
							trust = { who = ROOT value = 90 }
						}
						AND = {
							is_subject_of = ROOT
							trust = { who = ROOT value = 90 }
						}
						ROOT = {
							has_won_war_against = {
								who = PREV
								max_years_since = 50
							}
						}
						tag = ROOT
					}
				}
			}
		}
		required_missions = {
		}
		position = 9
		trigger = {
			all_country = {
				NOT = { great_power_rank = 9 }
				OR = {
					AND = {
						alliance_with = ROOT
						trust = { who = ROOT value = 90 }
					}
					AND = {
						is_subject_of = ROOT
						trust = { who = ROOT value = 90 }
					}
					ROOT = {
						has_won_war_against = {
							who = PREV
							max_years_since = 50
						}
					}
					tag = ROOT
				}
			}
		}

		effect = {
			add_country_modifier = {
				name = asur_ultimate_reward
				duration = -1
			}
		}
	}
}

gen_cadai_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		religion = cadai
	}


	gen_cadai_places_of_worship = {
		icon = mission_egyptian_mamluk_soldier
		provinces_to_highlight = {
			owned_by = ROOT
			NOT = { development = 6 }
		}

		required_missions = {
		}

		trigger = {
			all_owned_province = {
				development = 6
			}
			num_of_owned_provinces_with = {
				OR = {
					has_building = temple
					has_building = cathedral
				}
				value = 5
			}
		}

		effect = {

			add_country_modifier = {
				name = cadai_friends_with_cadai
				duration = -1
			}

		}
	}
