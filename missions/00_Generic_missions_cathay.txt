gen_yin_yang_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		religion = yin_yang
	}

	gen_yin_yang_happy_priest = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			estate_loyalty = {
			    estate = estate_church
			    loyalty = 65
			}
		}

		effect = { 
			add_country_modifier = { 
 				name = araby_happy_church
				duration = -1
			}
		}
	}

	gen_yin_yang_places_of_worship = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_yin_yang_happy_priest } 
		trigger = {

			num_of_owned_provinces_with = {

				OR = {
				 	has_building = temple
				 	has_building = cathedral
				}
				value = 5
			}
		}

		effect = { 
			change_innovativeness = 10
		}
	}

	gen_yin_yang_place_of_study = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_yin_yang_places_of_worship } 
		trigger = {
			capital_scope = {
				development = 30
			}
		}

		effect = { 
			capital_scope = {
				add_permanent_province_modifier = { 
 				  	name = araby_place_of_study
    				duration = -1  
				}
			}
		}
	}

	gen_yin_yang_places_of_law = { 
		icon = mission_egyptian_mamluk_soldier

		trigger = {
			num_of_owned_provinces_with = {

				OR = {
				 	has_building = courthouse

				}
				value = 5
			}
		}

		effect = { 
			add_country_modifier = { 
 				name = araby_place_of_law
				duration = -1
			}
		}
	}

	gen_yin_yang_places_of_innovation = {
