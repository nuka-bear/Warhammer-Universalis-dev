		}
		provinces_to_highlight = {
			OR = {

					tier = 0
				}

					type = great_sphinx_of_numasis
					tier = 0
				}
			}
		}
		trigger = {
			any_owned_province = {

					tier = 1
				}
			}

				has_great_project = {
					type = tomb_of_settra
					tier = 1
				}
			}
			any_owned_province = {
				has_great_project = {
					type = great_sphinx_of_numasis
					tier = 1
				}
			}
		}
		effect = {
			add_stability_or_adm_power = yes
			if = {
				limit = {
					adm = 6
				}
				add_adm_power = 100
			}
			else = {
				change_adm = 1
			}
			if = {
				limit = {
					dip = 6
				}
				add_dip_power = 100
			}
			else = {
				change_dip = 1
			}
			if = {
				limit = {
					mil = 6
				}
				add_mil_power = 100
			}
			else = {
				change_mil = 1
			}
		}
	}

	gen_nehekharan_pantheon_missions_4 = {
		icon = mission_tomb_king_religion
		required_missions = {
		}
		provinces_to_highlight = {
			OR = {
				has_great_project = {
					type = black_pyramid_of_nagash
					tier = 0
				}
				AND = {
					region = nehekhara_region
					NOT = {
						religion = ROOT
					}

			}
		}
		trigger = {
			NOT = {

			}
		}
