## Own the passes, Have centers of Trade at level 3, Rent out condeterii, Unite the tribes

gen_ogre_missions = {
	slot = 1
	generic = yes
	has_country_shield = yes
	ai = yes
	potential = {
		OR = {
			culture_group = ogre_group
			primary_culture = ogre_main
		}
	}

	gen_ogre_please_nobility = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			estate_loyalty = {
			    estate = estate_nobles
			    loyalty = 65
			}
			num_of_estate_agendas_completed = {
				estate = estate_nobles
				value = 1
			}
		}

		effect = {
			add_years_of_income = 1.0
			add_yearly_manpower = 1.0
		}
	}

	gen_ogre_unite_tribes = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_ogre_please_nobility } 
		trigger = {
			num_of_provinces_owned_or_owned_by_subjects_with = {
				value = 25
				superregion = mountains_of_mourn_superregion
			}
		}

		effect = {
			add_country_modifier = { 
 				name = ogre_claimant
				duration = -1
			}
		}
	}

	gen_ogre_mercenaries = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		trigger = {
			any_country = {
				ROOT = { is_renting_condottieri_to = PREV }
			}
		}

		effect = {
			add_years_of_income = 1.0
			add_yearly_manpower = 1.0
		}
	}


	gen_ogre_ivory_pass = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  } 
		position = 5
		provinces_to_highlight = {
			OR = {
				province_id = 4272
				province_id = 2812
				province_id = 2813
				province_id = 2814
				province_id = 2808
			}
		}
		trigger = {
			owns = 4272
			owns = 2812
			owns = 2813
			owns = 2814
			owns = 2808
		}

		effect = {
			add_country_modifier = { 
 				name = ogre_trade_1
				duration = -1
			}
		}
	}

	gen_ogre_silk_pass = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  }
		provinces_to_highlight = {
			OR = {
				province_id = 2818
				province_id = 2817
				province_id = 2802
				province_id = 2796
			}
		}
		trigger = {
			owns = 2818
			owns = 2817
			owns = 2802
			owns = 2796
		}

		effect = {
			add_country_modifier = { 
 				name = ogre_trade_2
				duration = -1
			}
		}
	}

	gen_ogre_north_pass = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = {  }
		provinces_to_highlight = {
			OR = {
				province_id = 2753
				province_id = 2759
				province_id = 2750
				province_id = 2754
				province_id = 2751
				province_id = 2752
			}
		}
		trigger = {
			owns = 2753
			owns = 2749
			owns = 2750
			OR = {
				owns = 2752
				AND = {
					owns = 2754
					owns = 2751
				}
			}
		}

		effect = {
			add_country_modifier = { 
 				name = ogre_trade_3
				duration = -1
			}
		}
	}


	gen_ogre_dominate_ivory_road = { 
		icon = mission_egyptian_mamluk_soldier
		required_missions = { gen_ogre_ivory_pass gen_ogre_silk_pass gen_ogre_north_pass} 
		provinces_to_highlight = {
			province_has_center_of_trade_of_level = 1
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				OR = {
					province_has_center_of_trade_of_level = 3 
				}
				value = 5 
			}
		}

		effect = {
			add_country_modifier = { 
 				name = gen_ogre_dominate_ivory_road
				duration = -1
			}
		}
	}
}
