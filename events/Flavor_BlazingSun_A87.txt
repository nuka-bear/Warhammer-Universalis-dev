#After First Internal Mission
country_event = {
	id = flavor_blazing_sun.2
	title = flavor_blazing_sun.2.t
	desc = flavor_blazing_sun.2.d
	picture = REFORM_eventPicture
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		tag = A87
	}
	immediate = {
	}
	#From this day Forward we shall become the Grand Order of the Blazing sun!
	option = {
		name = flavor_blazing_sun.2.a
		restore_country_name_effect = yes
		ai_chance = {
			factor = 100
		}
		if = {
			limit = {
				NOT = {
					government = theocracy
				}
			}
			change_government = theocracy
		}
		unlock_government_reform = {
			government_reform = grand_order_blazing_sun_reform
		}
		add_government_reform = grand_order_blazing_sun_reform
		if = {
			limit = {
				NOT = {
					government_rank = 2
				}
			}
			set_government_rank = 2
		}
		add_prestige = 25
	}
	#I think the Current Reform is good enough
	option = {
		name = flavor_blazing_sun.2.b
		change_government_reform_progress = 50
		unlock_government_reform = {
			government_reform = grand_order_blazing_sun_reform
		}
	}
}
