	id = event_everchosen.1
	title = event_everchosen.1.t
	desc = event_everchosen.1.d
	picture = ARTWORK_eventPicture
	
	
	is_triggered_only = no

		has_reform = everchosen_reform
		NOT = { is_at_war = yes }
	}

	# Declare War on Somebody

	option = {
		name = event_everchosen.1.option_1
		ai_chance = { factor = 10 }
		trigger = {
			any_neighbor_country = {
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
			}
		}
		add_yearly_manpower = 0.5

			LIMIT = {
				NOT = { is_subject_of = root }

				NOT = { ROOT = { truce_with = prev } }
			}

				declare_war_with_cb = {
		  			who = PREV

				}
			}
		}
	}

### Ruin the world

country_event = {
	id = event_everchosen.2
	title = event_everchosen.2.t
	desc = event_everchosen.2.d
	picture = ARTWORK_eventPicture
	
	
	is_triggered_only = no
	
	trigger = {
		has_reform = everchosen_reform
	}

	mean_time_to_happen = { months = 12 }
	
### Ruin
	option = {
		name = event_everchosen.2.option_1
		ai_chance = { factor = 10 }

		custom_tooltip = "event_everchosen.2.option_1.tt"

		hidden_effect = {

			every_owned_province = {

				if = {
					limit = {
						base_tax = 2
					}
					add_base_tax = -1
					controller = { add_treasury = 25 }
							
				}

				if = {
					limit = {
						base_production = 2
					}
					add_base_production = -1
					controller = { add_treasury = 25 }

				} 

				if = {
					limit = {
						base_manpower = 2
					}
					add_base_manpower = -1
					controller = { add_manpower = 1.0 }

				}
			}
		}
	}

### Better Not
	option = {
		name = event_everchosen.2.option_2
		ai_chance = { factor = 1 }
		random_owned_province = {
			spawn_rebels = {
				size = 1
				type = chaos_undivided_religion_rebels
			}
		}

		add_government_power = {
			mechanic_type = everchosen_mechanic
			power_type = everchosen_favor
			value = -2
		}


	}
}
