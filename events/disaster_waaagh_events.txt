#No Waagh Disaster Start
country_event = {
	id = disaster_no_waaagh.1
	title = disaster_no_waaagh.1.t
	desc = disaster_no_waaagh.1.d
	picture = ARTWORK_eventPicture
	
	
	is_triggered_only = yes
	
	trigger = {
	}
	

	# Declare War on Somebody

	option = {
		name = disaster_no_waaagh.1.option_1
		ai_chance = { factor = 100 }
		add_yearly_manpower = 0.5	
		random_neighbor_country = {
			LIMIT = {
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
			}

			ROOT = {
				declare_war_with_cb = {
		  			who = PREV
		    		casus_belli = cb_annex
				}
			}
		}
	}


	# Rebels instead

	option = {
		name = disaster_no_waaagh.1.option_2
		ai_chance = { factor = 10 }
		random_owned_province = {
			spawn_rebels = {
				size = 1
				type = pretender_rebels
			}
		}
	}
}
