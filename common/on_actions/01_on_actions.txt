### Raiding # FROM = Loser country?
on_siege_won_province = {
	if = {
		limit = {
			controller = {
				has_country_modifier = lets_raid_modifier
				NOT = { technology_group = FROM }
#				NOT = { owner = PREV }
			}

			NOT = { has_province_modifier = recently_raided }
			NOT = { has_influencing_fort = yes }
		}

		if = {
			limit = {
				base_tax = 2
			}
			add_base_tax = -1
			controller = { add_treasury = 25 }
					
			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
			}
			add_devastation = 10
			controller = { add_war_exhaustion = 0.05 }
		}

		if = {
			limit = {
				base_production = 2
			}
			add_base_production = -1
			controller = { add_treasury = 25 }

			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
				}
			add_devastation = 10	
			controller = { add_war_exhaustion = 0.05 }
		} 

		if = {
			limit = {
				base_manpower = 2
			}
			add_base_manpower = -1
			controller = { add_treasury = 25 }

			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
			}
			add_devastation = 10
			controller = { add_war_exhaustion = 0.05 }
		}
	}
}

#### Terrain & Race Bonus Transfer

on_province_owner_change = {

	remove_province_modifier = lizardfolk_mild_winter
	remove_province_modifier = lizardfolk_normal_winter
	remove_province_modifier = lizardfolk_severe_winter
	remove_province_modifier = lizardfolk_arctic
	remove_province_modifier = wood_elves_magical_forest
	remove_province_modifier = wood_elves_non_forest
	remove_province_modifier = undergrounder_underground
	remove_province_modifier = undergrounder_above_ground
	remove_province_modifier = surface_dweller_underground


## Lizardfolk
	if = {
		limit = {
			owner = { has_country_modifier = lizardmen_administration }
			has_climate = mild_winter
			}
		add_permanent_province_modifier = { 
			name = lizardfolk_mild_winter
			duration = -1  	
		}
	}

	if = {
		limit = {
			owner = { has_country_modifier = lizardmen_administration }
			has_climate = normal_winter
		}
		add_permanent_province_modifier = { 
		    name = lizardfolk_normal_winter
		    duration = -1  
		}
	}

	if = {
		limit = {
			owner = { has_country_modifier = lizardmen_administration }
			has_climate = severe_winter
		}
		add_permanent_province_modifier = { 
		    name = lizardfolk_severe_winter
		    duration = -1  
		}
	}

	if = {
		limit = {
			owner = { has_country_modifier = lizardmen_administration }
			has_climate = arctic
		}
		add_permanent_province_modifier = { 
		    name = lizardfolk_arctic
		    duration = -1  
		}
	}

### Wood Elves
	if = {
		limit = {
			owner = { technology_group = tech_wood_elf }
			has_terrain = magical_forest_terrain
		}
		add_permanent_province_modifier = { 
		    name = wood_elves_magical_forest
		    duration = -1  
		}
	}

	if = {
		limit = {
			owner = { technology_group = tech_wood_elf }
			NOT = { has_terrain = magical_forest_terrain }
			NOT = { has_terrain = forest }
			NOT = { has_terrain = jungle }
		}
		add_permanent_province_modifier = { 
		    name = wood_elves_non_forest
		    duration = -1  
		}
	}

### Dwarves and Skaven
	if = {
		limit = {
			OR = {
				owner = { technology_group = tech_dwarf }
				owner = { technology_group = tech_skaven }
			}
			OR = {
				has_terrain = subterranean_city_terrain
				has_terrain = subterranean_terrain
				has_terrain = urban_terrain
			}
		}	
		add_permanent_province_modifier = { 
		    name = undergrounder_underground
		    duration = -1  
		}
	}

	if = {
		limit = {
			OR = {
				owner = { technology_group = tech_dwarf }
				owner = { technology_group = tech_skaven }
			}
			NOT = { has_terrain = subterranean_city_terrain }
			NOT = { has_terrain = subterranean_terrain }
			NOT = { has_terrain = urban_terrain }
		}	
		add_permanent_province_modifier = { 
		    name = undergrounder_above_ground
		    duration = -1  
		}
	}

	if = {
		limit = {
			NOT = { technology_group = tech_dwarf }
			NOT = { technology_group = tech_skaven }
			NOT = { technology_group = tech_chaos_dwarfen }
			NOT = { technology_group = tech_hobgoblin }
			NOT = { technology_group = tech_orcish }
			NOT = { technology_group = tech_eastern_greenskin }
		}	
		add_permanent_province_modifier = { 
		    name = surface_dweller_underground
		    duration = -1  
		}
	}

}
