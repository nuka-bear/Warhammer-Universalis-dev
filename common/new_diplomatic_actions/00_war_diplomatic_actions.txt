# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

demote_dominion_action = {

	category = influence

	require_acceptance = no

	alert_index = 15

	alert_tooltip = tooltip_demote_dominion_alert_tooltip

	is_visible = {
		FROM = {
			is_subject_of_type_with_overlord = {
				who = ROOT
				type = dominion
			}
		}
	}

	is_allowed = {

		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		
			trust = {
				who = ROOT
				value = 90
			}

			owes_favors = { 
    	    	who = ROOT 
         		value = 50
			}

			NOT = { liberty_desire = 25 }
			is_at_war = no
		}

		ROOT = { is_at_war = no }

	}

    on_accept = {
    	vassalize = FROM

    	every_subject_country = {
			add_trust = {
			    	who = ROOT
			    	value = -15
			    	mutual = yes
			}
		}

		add_favors = {
			who = FROM
			amount = -50
		}
    }

    ai_will_do = {
        always = yes
    }
}


offer_dominion_action = {

	category = influence

	require_acceptance = yes

	alert_index = 15

	alert_tooltip = tooltip_offer_dominion_alert_tooltip

	is_visible = {
		FROM = {
			NOT = { is_subject = yes}
			NOT = { num_of_subjects = 1 }
			OR = {
				technology_group = ROOT
				religion = ROOT
			}
		}
	}

	is_allowed = {
		
		ROOT = { total_development = FROM }
		FROM = { alliance_with = ROOT }
		FROM = {
			has_opinion = {
				who = ROOT
				value = 100
			}
		}
		FROM = {
			owes_favors = { 
    	    	who = ROOT 
         		value = 90
			}
		
			trust = {
				who = ROOT
				value = 90
			}
		}
		FROM = { is_at_war = no }
		ROOT = { is_at_war = no }

	}

	ai_acceptance = {
        add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.125
			}
        }

		add_entry = {
			name = IN_HRE_OFFER
			limit = { FROM = { is_part_of_hre = yes } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -100 }
		}

		add_entry = {
			name = TOO_BIG_OFFER
			limit = { FROM = { total_development = ROOT } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -1000 }
		}

		add_entry = {
			name = OFFER_DIF_CULTURE
			limit = { NOT = { FROM = { primary_culture = ROOT } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -15 }
		}

		add_entry = {
			name = OFFER_DIF_TECH_GROUP
			limit = { NOT = { FROM = { technology_group = ROOT } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -50 }
		}

		add_entry = {
			name = OFFER_DIF_RELIGION
			limit = { NOT = { FROM = { religion = ROOT } } }

			multiply_variable = { which = ai_value value = -15 }
		}

				which = ai_value

			}

				which = ai_value
				value = 0.4
			}
		}

			name = AI_DIPLOMATIC_REPUTATION

				variable_name = ai_value
