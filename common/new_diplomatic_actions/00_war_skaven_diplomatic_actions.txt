kidnap_grey_seer = {

	category = covert

	require_acceptance = no

	is_visible = {
		religion = horned_rat
		FROM = { num_of_cardinals = 1 }
	}

	is_allowed = {

		has_spy_network_in = {
			who = FROM
			value = 100
		}

	}

    on_accept = {
    	FROM = {
    		random_owned_province = {
    			limit = { has_cardinal = yes }
    			remove_cardinal = yes
    		}

    		country_event = { 
			    id = skaven.1
			}
    	}

		random_owned_province = {
			limit = { has_cardinal = no}
			add_cardinal = yes
		}

		add_spy_network_in = {
		    who = FROM
		    value = -100
		}


    }

    ai_will_do = {
        factor = 1000
    }
}





ravage_borders = {

	category = covert

	require_acceptance = no

	is_visible = {
		has_country_modifier = skaven_administration
		is_neighbor_of = FROM
	}

	is_allowed = {

		has_spy_network_in = {
			who = FROM
			value = 50
		}

	}

    on_accept = {

    	random_neighbor_province = {
    		limit = {
    			owned_by = FROM
    			NOT = { devastation = 91 }
    			}
    		add_devastation = 10
    		ROOT = { add_treasury = 5 }
    	}

    	FROM = {
			add_opinion = {
			    who = ROOT
			    modifier = ravaged_border
			}
    	}

		add_spy_network_in = {
		    who = FROM
		    value = -50
		}

    }

    ai_will_do = {
    	factor = 1
    	modifier = {
    		factor = 0

	    	OR = {
		        FROM = { has_cardinal = yes }
	 			has_opinion = {
					who = FROM
					value = 25
				}
			}
		}       
    }

}
