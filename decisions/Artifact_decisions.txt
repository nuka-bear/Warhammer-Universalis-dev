### Resurrect Nagash

	resurrect_nagash_decision = {
		potential = {
			OR = {

				religion = vampiric
				religion = nagash_worship

				any_province = {
					owned_by = ROOT
					has_province_modifier = location_of_the_jade_blooded_vial
				}
				any_province = {
					owned_by = ROOT
					has_province_modifier = location_of_the_staff_of_nagash
				}
				any_province = {
					owned_by = ROOT
					has_province_modifier = location_of_the_carstein_ring
				}
			}
			all_country = { NOT = { has_country_modifier = nagash_resurrected_modifier } }
		 }

		allow = {

			religion = vampiric
			religion = nagash_worship

			any_province = {
				owned_by = ROOT
				has_province_modifier = location_of_the_jade_blooded_vial
			}
			any_province = {
				owned_by = ROOT
				has_province_modifier = location_of_the_staff_of_nagash
			}
			any_province = {
				owned_by = ROOT
				has_province_modifier = location_of_the_carstein_ring
			}

		manpower = 50
		treasury = 1000

		}

		effect = {

			add_manpower = -50
			add_treasury = -1000

			every_province ={
				limit = { has_province_modifier = location_of_the_jade_blooded_vial }
				remove_province_modifier = location_of_the_jade_blooded_vial
			}

			every_province ={
				limit = { has_province_modifier = location_of_the_staff_of_nagash }
				remove_province_modifier = location_of_the_staff_of_nagash
			}

			every_province ={
				limit = { has_province_modifier = location_of_the_carstein_ring }
				remove_province_modifier = location_of_the_carstein_ring
			}

			add_country_modifier = { 
			    name = nagash_resurrected_modifier
			    duration = -1  
			}

		}

		ai_will_do = { factor = 100 }
	}
